\newcommand\printNameDesc[6]{%
    \begingroup
        \begin{minipage}[t]{0.6\textwidth} 
            {\raggedright\huge\bfseries #1\par} 
            \vspace{0.3em} 
            {\raggedright\normalsize \href{https://#3}{#3}\par}
            \vspace{0.3em} 
            {\raggedright #2\par} 
        \end{minipage}%
        \begin{minipage}[t]{0.4\textwidth} 
            \raggedleft 
            #4 \par
            \href{mailto:#5}{#5} \par
            \href{https://#6}{#6}
        \end{minipage}
        \vspace{0.5em}
    \endgroup
}
