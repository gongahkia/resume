            {\raggedright\Large \href{https://#3}{#3}\par}
            \vspace{0.4em} 
            {\raggedright #2\par} 
        \end{minipage}%
        \begin{minipage}[t]{0.4\textwidth} 
            \raggedleft 
            #4 \par
            \href{mailto:#5}{#5} \par
            \href{https://#7}{#7} \par
            \href{https://#6}{#6}
        \end{minipage}
        \vspace{0.5em}
    \endgroup
}
